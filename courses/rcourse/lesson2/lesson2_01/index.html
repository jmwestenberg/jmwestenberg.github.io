require(tidyverse)                                                               # 1
require(xtable)                                                                  # 2
                                                                                 # 3
load(paste0(workingdir, file.path("processed_data", "county_data_all.rda")))     # 4
                                                                                 # 5
indiana<-county_data %>% filter(State_fips=="18")                                # 6
indiana_laborforce<-                                                             # 7
indiana %>% group_by(Year) %>%                                                   # 8
  summarise(`Min`=min(Labor_force),                                              # 9
    `10th`=quantile(Labor_force, c(0.1)),                                       # 10
    `25th`=quantile(Labor_force, c(0.25)),                                      # 11
    `50th`=quantile(Labor_force, c(0.5)),                                       # 12
    `75th`=quantile(Labor_force, c(0.75)),                                      # 13
    `90th`=quantile(Labor_force, c(0.9)),                                       # 14
    `Max`=max(Labor_force),                                                     # 15
    `Average`=mean(Labor_force),                                                # 16
    `St. Dev`=sd(Labor_force))                                                  # 17
                                                                                # 18
xtable(indiana_laborforce, type ="latex",                                       # 19
        digits=0, label= "tab:indiana_laborforce",                              # 20
        align = c("|c|", "|c|", "c", "c", "c", "c", "c", "c", "c", "c", "c|"))  # 21
